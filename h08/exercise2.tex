\begin{questions}
    \question
    Zwei Würfel werden unabhängig voneinander geworfen. 
    $X_i$ gebe die Augenzahl des $i$-ten Würfels ($i = 1,2$) an. 
    Für 
    \[ 
        X = X_1 + X_2 \quad \text{und} \quad Y = X_1 \cdot X_2
    \]
    sollen folgende Kenngrößen berechnet werden:
    \begin{parts}
        \part 
        Erwartungswert 
        \begin{solution}
            Offensichtlich gilt: 
            \[ 
                \mu = \Mean(X_1) = \Mean(X_2) = \frac{1}{6} \cdot \sum_{i=1}^6 i = \frac{7}{2} = 3.5 
            \]
            \[ 
                \sigma^2 = \Var(X_1) = \Var(X_2) = E((X_1 - \mu)^2) = \frac{1}{6} \cdot \sum_{i=1}^6 (i - 3.5)^2 = \frac{35}{12} \approx 2.9167
            \]
            und damit:
            \[ 
                \mu_X = E(X) = E(X_1) + E(X_2) = 7 \quad \land \quad \mu_Y = E(Y) = E(X_1) \cdot E(X_2) = \frac{49}{4} = 12.25
            \]
            \qed
        \end{solution}
        
        \part 
        Varianz        
        \begin{solution}
            Offensichtlich gilt: 
            \[ 
                X_1, X_2 \ \text{stochastisch unabhängig} \implies \Cov(X_1, X_2) = 0 
            \]
            Und damit: 
            \[ 
                \Var(X) = \Var(X_1) + \Var(X_2) + 2\Cov(X_1,X_2) = \frac{35}{6} \approx 5.833 
            \]
            \[ 
                \Var(Y) = \Mean(X_1)^2 \Var(X_2) + \Mean(X_2)^2 \Var(X_1) + \Var(X_1) \Var(X_2) = \frac{11515}{144} \approx 79.965
            \]
            \qed
        \end{solution}
        
        \newpage 
        \part 
        Kovarianz $\Cov(X,Y)$
        \begin{solution}
            Es gilt: 
            \begin{alignat*}{1}
                \Cov(X,Y) & = \Mean(XY) - \Mean(X) \Mean(Y)                                         \\ 
                          & = \Mean((X_1 + X_2)X_1X_2) - \Mean(X) \Mean(Y)                          \\ 
                          & = \Mean(X_1^2X_2 + X_1X_2^2) - \Mean(X) \Mean(Y)                        \\ 
                          & = \Mean(X_1^2) \Mean(X_2) + \Mean(X_1) \Mean(X_2^2) - \Mean(X) \Mean(Y) \\ 
                          & = \mu \left( \Mean(X_1^2) + \Mean(X_2^2) \right) - \Mean(X) \Mean(Y)    \\ 
                          & = \mu \left( 2\Mean(X_1^2) \right) - \Mean(X) \Mean(Y)                  \\ 
                          & = \mu \left( 2\sum_{i=1}^6 i^2 \cdot f_1(i) \right) - \Mean(X) \Mean(Y) \\ 
                          & = \frac{\mu}{3} \sum_{i=1}^6 i^2 - \Mean(X) \Mean(Y)                    \\ 
                          & = \frac{7}{6} \cdot 91 - 7 \cdot \frac{49}{4}                           \\ 
                          & = \frac{245}{12} \approx 20.4167
            \end{alignat*}
            
            \qed
        \end{solution}
        
        \part 
        Korrelationskoeffizient $\rho_{XY}$
        \begin{solution}
            Es gilt: 
            \[ 
                \rho_{XY} = \frac{\Cov(X,Y)}{\sigma_{X} \cdot \sigma_{Y}} = \frac{\frac{245}{12}}{\sqrt{\frac{35}{6}} \cdot \sqrt{\frac{11515}{144}}} = \frac{\sqrt{1974}}{47} \approx 0.9453
            \]
            \qed
        \end{solution}
    \end{parts}
